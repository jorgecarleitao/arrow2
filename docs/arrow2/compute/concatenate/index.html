Contains the concatenate kernel

Example:

use arrow2::array::Utf8Array;
use arrow2::compute::concatenate::concatenate;

let arr = concatenate(&[
    &Utf8Array::<i32>::from_slice(["hello", "world"]),
    &Utf8Array::<i32>::from_slice(["!"]),
]).unwrap();
assert_eq!(arr.len(), 3);
